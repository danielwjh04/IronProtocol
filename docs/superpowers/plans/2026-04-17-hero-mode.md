# Hero Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a toggleable "Hero Mode" gamification layer — a `UIModeContext` bus, `useCombatTrigger` hook, and `CombatCanvas` portal overlay — on top of the existing Focus Mode tracker without disrupting any DB or business logic.

**Architecture:** `UIModeProvider` wraps `App.tsx`; it holds `uiMode`, `setUIMode`, `dispatchCombat`, and `pendingBash`. `ActiveLogger` calls `useCombatTrigger().triggerBash(weight, reps)` after every set commit; the hook reads `uiMode` internally and no-ops in focus mode. `CombatCanvas` is a fixed portal rendered in `HeroOverlay`, which only mounts when `uiMode === 'hero'`.

**Tech Stack:** React 18, Framer Motion, Vitest, @testing-library/react, localStorage (no new dependencies)

import { createContext, useCallback, useContext, useEffect, useState } from 'react'
import type { ReactNode } from 'react'

type UIMode = 'focus' | 'hero'

interface PendingBash {
  intensity: number
  id: string
}

interface UIModeContextValue {
  uiMode: UIMode
  setUIMode: (mode: UIMode) => void
  dispatchCombat: (intensity: number) => void
  pendingBash: PendingBash | null
}

const LS_KEY = 'ip_ui_mode'

function readPersistedMode(): UIMode {
  try {
    return localStorage.getItem(LS_KEY) === 'hero' ? 'hero' : 'focus'
  } catch {
    return 'focus'
  }
}

const UIModeContext = createContext<UIModeContextValue | null>(null)

export function UIModeProvider({ children }: { children: ReactNode }) {
  const [uiMode, setUIModeState] = useState<UIMode>(readPersistedMode)
  const [pendingBash, setPendingBash] = useState<PendingBash | null>(null)

  useEffect(() => {
    document.documentElement.setAttribute('data-ui-mode', uiMode)
    document.documentElement.classList.add('pixel-fade')
    const t = setTimeout(() => document.documentElement.classList.remove('pixel-fade'), 500)
    return () => clearTimeout(t)
  }, [uiMode])

  const setUIMode = useCallback((mode: UIMode) => {
    setUIModeState(mode)
    try {
      localStorage.setItem(LS_KEY, mode)
    } catch {}
  }, [])

  const dispatchCombat = useCallback((intensity: number) => {
    setPendingBash({ intensity, id: crypto.randomUUID() })
  }, [])

  return (
    <UIModeContext.Provider value={{ uiMode, setUIMode, dispatchCombat, pendingBash }}>
      {children}
    </UIModeContext.Provider>
  )
}

export function useUIMode(): UIModeContextValue {
  const ctx = useContext(UIModeContext)
  if (!ctx) throw new Error('useUIMode must be used within UIModeProvider')
  return ctx
}
```

import { useUIMode } from '../context/UIModeContext'

export function useCombatTrigger() {
  const { uiMode, dispatchCombat } = useUIMode()

  function triggerBash(weight: number, reps: number): void {
    if (uiMode !== 'hero') return
    const intensity = Math.min((weight * reps) / 2000, 1)
    navigator.vibrate?.([intensity > 0.7 ? 120 : 60, 30, 40])
    dispatchCombat(intensity)
  }

  return { triggerBash }
}
```

import { useMemo } from 'react'
import { motion } from 'framer-motion'

interface Props {
  intensity: number
}

interface StarDatum {
  angle: number
  distance: number
}

export default function ImpactStars({ intensity }: Props) {
  const count = Math.max(1, Math.ceil(intensity * 12))

  const stars = useMemo<StarDatum[]>(
    () =>
      Array.from({ length: count }, (_, i) => ({
        angle: (i / count) * 360,
        distance: 40 + Math.random() * 60,
      })),
    [count],
  )

  const color = intensity > 0.7 ? '#FF3B3B' : '#3B71FE'

  return (
    <>
      {stars.map((star, i) => (
        <motion.div
          key={i}
          initial={{ opacity: 1, x: 0, y: 0, scale: 0 }}
          animate={{
            opacity: 0,
            x: Math.cos((star.angle * Math.PI) / 180) * star.distance,
            y: Math.sin((star.angle * Math.PI) / 180) * star.distance,
            scale: 0.5 + intensity * 1.5,
          }}
          transition={{ duration: 0.6, ease: 'easeOut' }}
          style={{
            position: 'absolute',
            width: 8,
            height: 8,
            borderRadius: '50%',
            background: color,
          }}
          aria-hidden="true"
        />
      ))}
    </>
  )
}

import { useEffect, useRef, useState } from 'react'
import { createPortal } from 'react-dom'
import { AnimatePresence, motion } from 'framer-motion'
import { useUIMode } from '../../context/UIModeContext'
import ImpactStars from './ImpactStars'

interface ActiveBash {
  intensity: number
  id: string
}

export default function CombatCanvas() {
  const { pendingBash } = useUIMode()
  const [activeBash, setActiveBash] = useState<ActiveBash | null>(null)
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null)

  useEffect(() => {
    if (!pendingBash) return
    setActiveBash(pendingBash)
    if (timerRef.current) clearTimeout(timerRef.current)
    timerRef.current = setTimeout(() => setActiveBash(null), 900)
    return () => {
      if (timerRef.current) clearTimeout(timerRef.current)
    }
  }, [pendingBash?.id])

  return createPortal(
    <div
      aria-hidden="true"
      style={{
        position: 'fixed',
        inset: 0,
        zIndex: 60,
        pointerEvents: 'none',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
      }}
    >
      <AnimatePresence>
        {activeBash && (
          <motion.div
            key={activeBash.id}
            aria-label="combat sequence"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            style={{ position: 'relative', display: 'flex', alignItems: 'center', gap: 40 }}
          >
            {/* Fighter placeholder — swap this SVG child for a real sprite sheet */}
            <motion.svg
              width="48"
              height="64"
              viewBox="0 0 48 64"
              animate={{ x: [0, 30, 0] }}

              aria-label="fighter"

              <rect x="12" y="20" width="24" height="30" rx="4" fill="#3B71FE" />
              <rect x="4" y="22" width="10" height="4" rx="2" fill="#3B71FE" />
            </motion.svg>

            {/* Impact origin — ImpactStars burst outward from this point */}
            <div style={{ position: 'relative', width: 0, height: 0 }}>

            </div>
